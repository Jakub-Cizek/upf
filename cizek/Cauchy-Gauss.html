def Cauchy_pdf(x):
    return(1/np.pi*1/(1+x**2))

def Gaussian_pdf(x,mu,sigma):
    return(1/(np.sqrt(2*np.pi)*sigma)*np.exp(-(x-mu)**2/(2*sigma**2)))

N=1000 #celkov poet dat
mean_C=np.empty(N) #aritmentick prmr hodnot z Cauchyho rozdlen
mean_G=np.empty(N) #aritmentick prmr hodnot z Normlnho rozdlen N(0,0.85)
median_C=np.empty(N) #medin hodnot z Cauchyho rozdlen
median_G=np.empty(N) #medin hodnot z Normlnho rozdlen N(0,0.85)
x_C=np.empty(N) #pole hodnot z Cauchyho rozdlen
x_G=np.empty(N) #pole hodnot z Normlnho rozdlen N(0,0.85)
for i in range(N):
    x_C[i]=np.tan(np.pi*(np.random.random()-0.5)) #simulace hodnot z Cauchyho rozdlen
    x_G[i]=np.random.normal(0,0.85) #simulace hodnot z Normlnho rozdlen N(0,0.85)
    mean_C[i]=np.mean(x_C[0:i])
    mean_G[i]=np.mean(x_G[0:i])
    median_C[i]=np.median(x_C[0:i])
    median_G[i]=np.median(x_G[0:i])

import matplotlib.pyplot as plt

N=100000 #poet simulovanch hodnot pro histogram
Nk=np.arange(1,10,1)  #kolik hodnot z rozdlen pro matematick kyvadlo stat
y=np.zeros(N)
for k in Nk:
    for i in range(N):
        for j in range(k):
            y[i]=y[i]+np.sin(np.pi*(np.random.random()-0.5))
    plt.hist(y,bins=100,density=True)
    plt.title('souet '+str(k)+' hodnot') 
    plt.show()

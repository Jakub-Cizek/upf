import numpy as np
import matplotlib.pyplot as plt
In[10]:
Nmax=10000 #maximln poet krok
p=0.5 #pravdpodobnost, e udlm krok vpravo
N_repeat=1000 #poet opakovn simulac
N=np.arange(0,Nmax+1,1) #pole celkovch pot krok
mean_distance=np.zeros(np.size(N)) #prmrn vzdlenost od stedu 
mean_distance_abs=np.zeros(np.size(N)) #prmrn vzdlenost od stedu 
for i in N:
    data=np.random.binomial(N[i],p,N_repeat) #simulace N_repeat hodnot z binomick0ho rozdlen B(N,p)
    mean_distance[i]=np.sqrt(np.mean((2*data-N[i])**2)) #vpoet prmrn hodnoty vzdlenosti od stedu jako sqrt(<(2k-N)**2>)
    mean_distance_abs[i]=np.mean(abs((2*data-N[i]))) #vpoet prmrn hodnoty vzdlenosti od stedu jako |2k-N|
In[22]:
plt.plot(N,mean_distance,label='simulace: odmocnina z kvadrtu souadnice')
plt.plot(N,mean_distance_abs,c='green', label='simulace: absolutn hodnota')
plt.plot(N,np.sqrt(N),c='red',label='teorie: odmocnina z kvadrtu souadnice')
plt.plot(N,np.sqrt(2*N/np.pi),c='violet',label='teorie: absolutn hodnota')
plt.xlim(0,Nmax)
plt.ylim(0,)
plt.xlabel('poet krok')
plt.ylabel('vzdlenost')
plt.legend()
plt.show()
